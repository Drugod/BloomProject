ACTOR PossesedShotgunguy: BloomPossessed_Base 19089
{
	//$Title Possesed Shotgunguy	
	SeeSound "Moron/Cry1"
	AttackSound "shotguy/attack"
	DeathSound "Moron/Crying"
	ActiveSound "Moron/GoAway"
	PainSound "Moron/Scream"
	States
	{
	Spawn:
		HAND GGGGGHHHHHIIIIIJJJJJ 1 A_Look
		Loop
	See:
		HAND AAAABBBBCCCCDDDDEEEEFFFFAAAABBBBCCCCDDDDEEEEFFFFAAAABBBBCCCCDDDDEEEEFFFFAAAABBBCCCCDDDDEEEEFFFF 1 A_Chase
		TNT1 A 0 A_Jump(64,"Flee")
		Loop
	Flee:
		HAND AAAABBBBCCCCDDDDEEEEFFFFAAAABBBBCCCCDDDDEEEEFFFFAAAABBBBCCCCDDDDEEEEFFFFAAAABBBCCCCDDDDEEEEFFFF 1 A_Wander
		TNT1 A 0 A_Jump(200,"Stand")
		Loop
	Stand:
		HAND GGGGGHHHHHIIIIIJJJJJ 1
		Goto See
	Missile:
		HAND K 10 A_FaceTarget
		HAND L 10 BRIGHT A_SPosAttackUseAtkSound
		HAND K 10
		TNT1 A 0 A_Jump(160,"Flee")
	LaunchTnt:
		HATN A 5 A_FaceTarget
		HATN BC 5
		HATN A 0 A_ThrowGrenade("DoomMonsterTNT",0,20,4,0)
		HATN D 5
		Goto See
	Pain:
		HAND M 4 A_Pain
		Goto Flee
	Death:
		SPOS H 5
		SPOS I 5 A_Scream
		SPOS J 5 A_NoBlocking
		SPOS K 5
		SPOS L -1
		Stop
	XDeath:
		SPOS M 5 A_SpawnDebris("FlyingBlood")
		TNT1 A 0 A_SpawnDebris ("EyeGuts")
		TNT1 A 0 A_SpawnDebris ("Bloodliver")
		TNT1 A 0 A_SpawnDebris ("HeadBlood")
		TNT1 A 0 A_SpawnDebris ("Hand1")
		TNT1 A 0 A_SpawnDebris ("Hand1")
		TNT1 A 0 A_SpawnItemEx ("Intestines",random(-50,50),random(-50,50),0)
		SPOS N 5 A_XScream
		TNT1 A 0 A_SpawnDebris ("OrganDebris2")
		TNT1 A 0 A_SpawnDebris ("Bloodliver")
		TNT1 A 0 A_SpawnDebris("FlyingBlood")
		SPOS O 5 A_NoBlocking
		TNT1 A 0 A_SpawnDebris ("OrganDebris")
		TNT1 A 0 A_SpawnDebris ("Bloodliver")
		TNT1 A 0 A_SpawnDebris("FlyingBlood")
		SPOS PQRST 5 A_SpawnDebris("FlyingBlood")
		TNT1 A 0 A_SpawnItemEx ("Entrails",random(-50,50),random(-50,50),0)
		TNT1 A 0 A_SpawnItemEx("Blood_",random(-20,20),random(-20,20),random(-20,20),random(0,15),random(0,15),random(0,15),random(0,360),0,120)
		SPOS U -1
		Stop
	Raise:
		SPOS L 5
		SPOS KJIH 5
		Goto See
	Death.Fire:
	Death.Burn:
	Death.Flare:
	Burn:
		"####" A 1 A_JumpIf(waterlevel > 1, "XDeath")
		TNT1 A 0 A_Scream
		TNT1 A 0 A_NoBlocking
		TNT1 A 0 A_Jump(128,"AltBurn")
		TNT1 A 0 A_SpawnItem("BurnFormerHuman", 0, 7)
		Stop
	AltBurn:
		BURN A 3 BRIGHT A_PlaySound ("human/imonfire")
		TNT1 A 0 A_SpawnItem("BFireTrail",8+random(-5,5),20+random(-5,5))
		BURN B 3 BRIGHT A_SpawnItem("BFireTrail",8+random(-5,5),20+random(-5,5))
		TNT1 A 0 A_SpawnItem("BFireTrail",8+random(-5,5),20+random(-5,5))
		BURN C 3 BRIGHT A_Wander
		TNT1 A 0 A_SpawnItem("BFireTrail",8+random(-5,5),20+random(-5,5))
		BURN D 3 BRIGHT A_NoBlocking
		TNT1 A 0 A_SpawnItem("BFireTrail",8+random(-5,5),20+random(-5,5))
		BURN E 5 BRIGHT A_SpawnItem("BFireTrail",8+random(-5,5),20+random(-5,5))
		TNT1 A 0 A_SpawnItem("BFireTrail",8+random(-5,5),20+random(-5,5))
		BURN FGH 5 BRIGHT A_Wander
		BURN I 5 BRIGHT A_SpawnItem("BFireTrail",8+random(-5,5),20+random(-5,5))
		BURN JKL 5 BRIGHT A_Wander
		BURN M 5 BRIGHT A_SpawnItem("BFireTrail",8+random(-5,5),20+random(-5,5))
		BURN N 5 BRIGHT A_SpawnDebris ("NapalmDebris")
		BURN OPQPQ 5 BRIGHT A_SpawnDebris ("NapalmDebris")
		BURN RSTU 7 BRIGHT A_SpawnDebris ("NapalmDebris")
		TNT1 A 0 A_SpawnItemEx ("BigSmoke",0,0,0)
		BURN V -1
		Stop
	}
}
