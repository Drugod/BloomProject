{
	//$Title Gargoyle's Soul
	Health 100
	Radius 16
	Mass 50
	Damage 3
	PainChance 256
	+DONTFALL
	+FLOAT
	+MISSILEMORE
	+NOGRAVITY
	+NOICEDEATH
	AttackSound "skull/Melee"
	PainSound "skull/pain"
	DeathSound "skull/death"
	ActiveSound "skull/active"
	RenderStyle SoulTrans
	Obituary "$OB_GSOUL"
	Species "Cacos"
	States
	{
	Spawn:
		GKUL AB 10 A_Look
		Loop
	See:
		GKUL AB 6 A_Chase
		Loop
	Missile:
		GKUL C 10 A_FaceTarget
		GKUL D 4 A_SkullAttack
		GKUL CD 4
		Goto Missile+2
	Pain:
		GKUL E 3
		GKUL E 3 A_Pain
		Goto See
	Death:
		GKUL F 6
		GKUL G 6 A_Scream
		GKUL H 6
		GKUL I 6 A_NoBlocking
		GKUL J 6
		GKUL K 6
		Stop
	Ice:
		GKUF R 5 A_FreezeDeath
		GKUF R 1 A_FreezeDeathChunks
		Wait
	}
}
