//========== KEYS ==========//
ACTOR Key1 : BlueCard replaces BlueCard
{
	Scale 0.5
	Inventory.PickupSound "PICKKEY"
	Inventory.PickupMessage "$DAGKEY"
	States
	{
	Spawn:
		TNT1 A 0 NODELAY A_JumpIf(CallACS("blm_voxmod")==1,"ModelSpawn")
	BaseSpawn:
		DKEY A -1
		Stop
	ModelSpawn:
		UNKN A -1
		Stop
	}
}
