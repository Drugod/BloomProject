	Tag "Pitchfork"
	Obituary "$OB_FORK"
	Weapon.Kickback 200
	+INVENTORY.UNDROPPABLE
	+NODAMAGETHRUST
	+THRUGHOST
	+WEAPON.MELEEWEAPON
	+WEAPON.NOALERT

	States
	{
	Spawn:

		Stop

		TOMG A 0 ACS_NamedExecuteAlways("TeslaCounter", 0)
		TNT1 A 0 A_JumpIfInventory("RemoveProximity", 1, "TakeProximity")

		TNT1 A 0 A_JumpIfInventory("RemoveTNT", 1, "TakeTNT")
		TNT1 A 0 A_JumpIfInventory("RemoveVodoo", 1, "TakeDoll")

		TNT1 A 0 A_TakeInventory("UsingPitchfork", 1)
		TNT1 AA 0 A_Lower
		PITC A 1 A_Lower
		Goto Deselect+1

		TNT1 AA 0 A_Raise
		PITC A 1 A_Raise
		Loop
	Fire:
		PITC BBB 1 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		PITC CC 1 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		TNT1 A 0 A_FireBullets(14, 0, 4, 4, "PitchforkPuff",1,64)
		PITC DDDCCCBBBAA 1 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		TNT1 A 0 A_ReFire
		Goto Ready+6
	TakeSpray:
		TNT1 A 0 A_TakeInventory("SprayCan", 1)
		TNT1 A 0 A_takeInventory("RemoveSpray", 1)
		Goto Ready
	TakeTNT:
		TNT1 A 0 A_TakeInventory("Dynamite", 1)
		TNT1 A 0 A_TakeInventory("RemoveTNT", 1)
		Goto Ready
	TakeDoll:
		TNT1 A 0 A_TakeInventory("VoodooDoll", 1)
		TNT1 A 0 A_TakeInventory("RemoveVodoo", 1)
		Goto Ready
	TakeProximity:
		TNT1 A 0 A_TakeInventory("ProximityDynamite", 1)
		TNT1 A 0 A_TakeInventory("RemoveProximity", 1)
		Goto Ready
	}
}
