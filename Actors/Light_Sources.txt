ACTOR FlameCult 3236
{
  //$Category "BlooM Light Sources"
  Radius 16
  Height 54
  Speed 12
  Scale 0.8
  Alpha 0.8
  RenderStyle Translucent
  Obituary "%o burned by cult fire!"
  Damage (random(10,20))
  Projectile
  +CANNOTPUSH
  +NODAMAGETHRUST
  +RIPPER
  +BLOODLESSIMPACT
  DamageType "Fire"
  States
  {
  Spawn:
	TNT1 A 0 A_JUMP (50,"SPAWN1")
	TNT1 A 0 A_JUMP (50,"SPAWN2")
	TNT1 A 0 A_JUMP (50,"SPAWN3")
	TNT1 A 0 A_JUMP (50,"SPAWN4")
	TNT1 A 0 A_JUMP (50,"SPAWN5")
  Loop
  Spawn1:
		CVLT ABCDEFGHIJKLMN 4 BRIGHT
		TNT1  A 0 A_Playsound("CVLTFIR")
		goto spawn
  Spawn2:
		CVLT FGHIJKLMNABCDE 4 BRIGHT
		TNT1  A 0 A_Playsound("CVLTFIR")		
		goto spawn
  Spawn3:
		CVLT JKLMNABCDEFGHI 4 BRIGHT
		TNT1  A 0 A_Playsound("CVLTFIR")		
		goto spawn
  Spawn4:
		CVLT LMNABCDEFGHIJK 4 BRIGHT
		TNT1  A 0 A_Playsound("CVLTFIR")		
		goto spawn		
  Spawn5:
		CVLT EFGHIABCDJKLMN 4 BRIGHT
		TNT1  A 0 A_Playsound("CVLTFIR")		
		goto spawn
  Death:
	TNT1 A 0 A_JUMP (50,"SPAWN1")
	TNT1 A 0 A_JUMP (50,"SPAWN2")
	TNT1 A 0 A_JUMP (50,"SPAWN3")
	TNT1 A 0 A_JUMP (50,"SPAWN4")
	TNT1 A 0 A_JUMP (50,"SPAWN5")
  Stop
  }
}
