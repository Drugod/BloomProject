ACTOR HandGrenades : DoomW
{
	Weapon.AmmoUse1 1
	Weapon.AmmoGive1 1
	Weapon.AmmoUse2 1
	Weapon.AmmoGive2 0
	Weapon.SelectionOrder 99999
	Weapon.AmmoType1 "GrenadeAmmo"
	Weapon.AmmoType2 "GrenadeAmmo"
	Obituary "$OB_FRAG"
    AttackSound "None"
    Inventory.PickupSound "AmmoAPIN"
	Inventory.PickupMessage "$FRAGPICK"
    +WEAPON.NOAUTOAIM
    +WEAPON.NOALERT
	+WEAPON.NOAUTOFIRE
    Scale 0.3
	States
	{
	Spawn:
		TNT1 A 0 NODELAY A_JumpIf(CallACS("blm_voxmod")==1,"ModelSpawn")
	BaseSpawn:
		GRND H 1
		GRND H -1 A_SpawnItemEx("HappyBundle",0,0,0)
		Stop
	ModelSpawn:
		UNKN A 1
		UNKN A -1 A_SpawnItemEx("HappyBundle",0,0,0)
		Stop
	Ready:
		TNT1 A 0 A_TakeInventory("ThrowPower", 150)
	    TNT1 A 0
		TNT1 A 0 A_JumpIfInventory("GrenadeAmmo", 1, 1)
		Goto NoGrenades
		//GRHO AB 1
		HNDG AB 1
		Goto Steady
	Steady:
		HNDG A 1 A_WeaponReady
		Loop
	NoGrenades:
		TNT1 A 1 A_WeaponReady
		Goto Ready
	Deselect:
		TNT1 A 0
		TNT1 A 0 A_JumpIfInventory("FiredGrenade", 1, "DropGrenade")
		TNT1 A 1 A_Lower
		Wait
	DropGrenade:
		TNT1 A 0
		TNT1 A 0 A_TakeInventory("FiredGrenade", 1)
		TNT1 A 0 A_FireCustomMissile("HandGrenadeDroped", frandom(-2,2), 1, 0, 0, 0, 0)
		Goto Deselect
	Select:
		HNDG A 1 A_Raise
		Loop
	Fire:
	    HNDG ABCD 1
		TNT1 A 0 A_GiveInventory("FiredGrenade", 1)
		TNT1 A 0 A_PLaySound ("grenades/grnpin")
		HNDG EEFG 1
	Hold:
	    TNT1 A 1 A_GiveInventory("ThrowPower", 1)
		TNT1 A 1 A_Refire
		TNT1 A 0 A_PLaySound ("grenades/grntoss")
		HNDG HI 1
		TNT1 A 0 A_FireCustomMissile("HandGrenade", frandom(-2,2), 1, 0, 0, 0, 0)
	    TNT1 A 0 A_TakeInventory("FiredGrenade", 1)
		HNDG JKLMNOPQ 3
		TNT1 A 3
		Goto Ready
	AltFire:
	    HNDG ABCD 1
		TNT1 A 0 A_GiveInventory("FiredGrenade", 1)
		TNT1 A 0 A_PLaySound ("grenades/grnpin")
		HNDG EEFG 1
	AltHold:
	    TNT1 A 1
		TNT1 A 1 A_Refire
		TNT1 A 0 A_PLaySound ("grenades/grntoss")
		HNDG HI 1
		TNT1 A 0 A_FireCustomMissile("HandGrenadeAlt", 0, 1, 0, 0, 0, 0)
	    TNT1 A 0 A_TakeInventory("FiredGrenade", 1)
		HNDG JKLM 1
		HNDG NOPQ 1
		TNT1 A 3
		Goto Ready
	TossGrenade:
		TNT1 A 0
		TNT1 A 0 A_TakeInventory("TossGrenade", 1)
		TNT1 A 0 A_JumpIfInventory("GrenadeAmmo", 1, 1)
		Goto NoGrenade
		HNDG ABCD 1
		TNT1 A 0 A_GiveInventory("FiredGrenade", 1)
		TNT1 A 0 A_PLaySound ("grenades/grnpin")
		HNDG EEFG 1
		TNT1 A 0 A_PLaySound ("grenades/grntoss")
		HNDG HI 1
		TNT1 A 0 A_TakeInventory("GrenadeAmmo", 1)
		TNT1 A 0 A_FireCustomMissile("HandGrenade", frandom(-2,2), 0, 0, 0, 0, 0)
	    TNT1 A 0 A_TakeInventory("FiredGrenade", 1)
		HNDG JKLM 1
		HNDG NOPQ 1
		TNT1 A 1
		TNT1 A 0 A_TakeInventory("TossGrenade", 1)
		Goto Ready
	NoGrenade:
		TNT1 A 0
		TNT1 A 0 A_Print("No Grenades Left.")
		Goto Ready+8
	}
}

ACTOR HandGrenade
{
	Radius 4
	Height 4
	Projectile
	Speed 30
	Damage 0
	Gravity 0.7
	Scale 0.2
	Projectile
	+DOOMBOUNCE
	+EXPLODEONWATER
	+MISSILE
	+SKYEXPLODE
	-BLOODSPLATTER
	-EXTREMEDEATH
	-NOGRAVITY
	BounceFactor 0.5
	WallBounceFactor 0.25
	Health 5
	SeeSound "grenades/grnbnce"
	DeathSound "none"
	Obituary "$OB_HAND"
	States
	{
	Spawn:
		HGRN ABCDEF 2
		TNT1 A 0 A_GiveInventory("GrenadeTime", 1)
		TNT1 A 0 A_JumpIfInventory("GrenadeTime", 3, "Explode")
		Loop
	Death:
		TNT1 A 0
		HGRN F 16
		TNT1 A 0 A_GiveInventory("GrenadeTime", 1)
		TNT1 A 0 A_JumpIfInventory("GrenadeTime", 3, "Explode")
		Loop
	Explode:
		TNT1 A 0 A_NoBlocking
		TNT1 A 0 A_CustomMissile("FloorKaboom",0,0,0,CMF_TRACKOWNER)
		TNT1 A 0 A_Explode(200, 150, XF_HURTSOURCE)
		TNT1 A 0 A_Playsound("MediumExplosion")
		TNT1 A 0 Radius_Quake (2, 24, 0, 15, 0)
		TNT1 A 0 A_AlertMonsters
		TNT1 A 0 A_SpawnItemEx ("BigSmoke",0,0,0)
		Stop
	}
}
