//WATER//
ACTOR WaterParticleX : ParticleBase
{
	States
	{
	Spawn:
		TNT1 A 1 
		TNT1 A 0 A_ChangeFlag(DOOMBOUNCE, 0)
		TNT1 A 1
		Goto Stand
	Stand:
		BSPH ABCD 4
		Stop
	Death:
		TNT1 A 1
		TNT1 A 0
		Stop
	}
}
